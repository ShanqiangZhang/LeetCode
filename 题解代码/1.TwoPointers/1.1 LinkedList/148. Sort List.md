题目:
    sort list
解法:
    1.暴力复制
    2.归并排序

1.暴力复制
class Solution {
    public ListNode sortList(ListNode head) {
        if (head == null || head.next == null) return head;
        List<Integer> list = new ArrayList<>();
        ListNode p1 = head;
        while (p1 != null) {
            list.add(p1.val);
            p1 = p1.next;
        }
        Collections.sort(list);
        p1 = head;
        for (int n : list) {
            p1.next = new ListNode(n);
            p1 = p1.next;
        }
        return head.next;
    }
}
复杂度:
    Time: O(nlogn) sort最花时间
    Space: O(n)

2.归并排序
class Solution {
    public ListNode sortList(ListNode head) {
        if (head == null || head.next == null) return head;
        //1. find mid 
        ListNode mid = findMid(head);
        //2.split into two lists
        ListNode head2 = mid.next;
        mid.next = null;
        //3.recusrive to split until there are only two nodes left
        ListNode left = sortList(head), right = sortList(head2);
        //4.merge list
        return merge(left, right);
    }

    public ListNode findMid(ListNode head) {
        ListNode dummy = new ListNode(-1, head), f = dummy, s = dummy;
        while (f != null && f.next != null) {
            f = f.next.next;
            s = s.next;
        }
        return s;
    }

    public ListNode merge(ListNode l1, ListNode l2) {
        ListNode dummy = new ListNode(-1), p1 = dummy;
        while (l1 != null && l2 != null) {
            if (l1.val < l2.val) {
                
                p1.next = l1;
                l1 = l1.next;
            } else {
                p1.next = l2;
                l2 = l2.next;
            }
            p1 = p1.next;
        }
        p1.next = l1 == null ? l2 : l1;
        return dummy.next;
    }
}

复杂度:
    Time: O(n)
    Space: O(logn) 递归栈空间
